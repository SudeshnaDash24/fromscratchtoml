X1 = Distribution.linear(pts=50,
               mean=[8, 20],
               covr=[[1.5, 1], [1, 2]])
X2 = Distribution.linear(pts=50,
               mean=[8, 15],
               covr=[[1.5, -1], [-1, 2]])

X3 = Distribution.linear(pts=50,
               mean=[15, 20],
               covr=[[1.5, 1], [1, 2]])
X4 = Distribution.linear(pts=50,
               mean=[15, 15],
               covr=[[1.5, -1], [-1, 2]])

X1 = ch.cat((X1, X2))
X2 = ch.cat((X3, X4))

Y1 = ch.ones(X1.size()[0])
Y2 = -ch.ones(X2.size()[0])

X = ch.cat((X1, X2))
y = ch.cat((Y1, Y2))

X = ch.stack([X1, X2])

binary_visualize(X)
