n = 10
n_dim = 3
nc = 3

X = np.zeros([n, n_dim])
for i in range(n):
    X[i,2] = i
    X[i,1] = 5*i + 2
#     X[i,1] = pow(i,2)
    X[i,0] = (5 - 2*X[i,0] - 3*X[i,1])/2
